export const Highlight = ({children, color, url, style={}}) => (
  <a style={{...style, ...{color: color, fontWeight: 'bold', textDecoration: 'none'}}} href={url}>
    {children}
  </a>
);

export const TitleImage = ({children, image, size, style = {}}) => (
  <div style={{display: 'inline-flex', alignItems: 'center'}}>
    <img src={image} style={{...style,...{width: 'auto', height: size, marginRight: '8px', marginTop: '3px'}}} />
    <div style={{fontSize: size, fontWeight: 'bold'}}>{children}</div>
  </div>
)

export const InlineImage = ({children, image, size, style = {}, text = ""}) => {
